// Operadores de Comparação //

> maior que
< menor que
>= maior igual que
<= menor igual que
=== igualdade estrita (valor e tipo)
== igualdade (valor)
!= diferente (valor)
!== diferente estrito (valor e tipo)

// Operadores Lógicos //
&& -> AND -> E
|| -> OR -> OU
! -> NOT -> NÃO


// Avaliação de curto circuíto //

    false
    0
    ''
    ""
    ``
    null
    undefined
    NaN
